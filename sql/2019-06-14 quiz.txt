-- 1. 고객 구매금액을 기준으로 내림차순으로 정렬해서 이름, 아이디, 생년월일, 총구매금액을 출력하시오

select cust_nm, cust_id, cust_birth, sales_amt*sales_cnt as '총구매금액' 
	from perf
	order by 총구매금액 desc;

-- 2. 고객 연령과 방문횟수를 비교하시오

select 2019-left(cust_birth, 4) as '고객연령', visit_cnt from perf;

select cust_birth, visit_cnt from perf;



-- 3. 방문횟수가 가장 많은 고객의 이름을 출력하시오

select cust_nm from perf 
	order by visit_cnt desc limit 1;
    
select cust_nm from perf
	where visit_cnt = (select max(visit_cnt) from perf);

-- 4. 1980년 이전 출생한 고객을 출력하시오.
select * from perf where cust_birth<='1980-01-01';

-- 5. 평균 방문횟수, 평균구입금액, 평균 구입상품 수를 구하시오

select avg(visit_cnt) as '평균 방문횟수', 
	avg(sales_amt*sales_cnt) as '평균 구입금액',
    avg(visit_cnt) as '평균 구입상품 수'
		from perf;

-- 6. 방문 당 구매금액이 가장 큰 고객의 이름을 구하시오
    
select cust_nm, sales_amt/visit_cnt as '방문당평균구매금액' 
	from perf
    order by 방문당평균구매금액 desc limit 1;

select cust_nm from perf
	where sales_amt/visit_cnt =
    (select max(sales_amt/visit_cnt) from perf);
